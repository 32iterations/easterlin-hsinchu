### 4.2 需求與供給配對演算法

```python
# 簡化版配對邏輯

def match_volunteers(request):
    """
    請求:
    ├─ 需求方: 竹科媽媽, 需要國中數學家教
    ├─ 人次: 每週2次, 90分鐘
    ├─ 時間: 周三/五 19:00-20:30
    └─ 地點: 赤土崎館 4F教室
    """

    # 1. 篩選候選志工
    candidates = search(
        skill = "數學教學",
        level = ≥ 60歲 or 高中以上
    )  # 符合: 30人

    # 2. 評估時間相容性
    for candidate in candidates:
        time_score = calculate_overlap_hours(
            volunteer_available_hours,
            request_time_slots
        )  # 周三/五 19:00-20:30 有空
        # 篩選到: 15人

    # 3. 計算技能匹配度 (使用向量相似度)
    skills_embedding = bert_model.encode(
        "資深教師" vs "高中數學" vs "教學經驗10年"
    )
    skill_match_score = cosine_similarity(
        request_skill_embedding,
        candidate_skill_embedding
    )  # 範圍 0-1

    # 4. 考慮地理位置
    distance_score = 1 / (1 + distance_km)
    # 近郊: 得分高 (鼓勵在地連結)

    # 5. 綜合評分
    final_score = (
        0.5 * skill_match_score +  # 技能最重要
        0.3 * time_score +         # 時間次重要
        0.2 * distance_score       # 距離加分
    )

    # 6. 排序與推薦
    top_3_matches = sorted(candidates,
                          key=final_score,
                          reverse=True)[:3]

    # 推薦:
    # #1: 王阿媽 (退休數學老師, 評分0.92)
    # #2: 李伯伯 (工程師, 數學出身, 評分0.88)
    # #3: 陳姐姐 (高中老師, 評分0.85)

    return top_3_matches
```
